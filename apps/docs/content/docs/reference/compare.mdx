# Compare Utilities

Compare utilities provide functions for comparing dates and timestamps, including relative time comparisons and date equality checks.

## isToday

Check if a date is today.

```typescript
function isToday(date: OptionsOrTimestamp, timeZone: TimeZone): boolean
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { isToday } from '@/datezone';

const today = new Date().getTime();
const yesterday = new Date(Date.now() - 86400000).getTime();

console.log(isToday(today, 'UTC')); // true
console.log(isToday(yesterday, 'UTC')); // false
```

</Tab>
<Tab value="Timezone">

```typescript
import { isToday } from '@/datezone';

const timestamp = new Date().getTime();
const isTodayNY = isToday(timestamp, 'America/New_York');
const isTodayTokyo = isToday(timestamp, 'Asia/Tokyo');

console.log(isTodayNY); // true/false depending on NY timezone
console.log(isTodayTokyo); // true/false depending on Tokyo timezone
```

</Tab>
</Tabs>

## isBefore

Check if the first date is before the second date.

```typescript
function isBefore(date1: number, date2: number): boolean
function isBefore(date1: DateOptions, date2: DateOptions): boolean
function isBefore(date1: OptionsOrTimestamp, date2: OptionsOrTimestamp, timeZone: TimeZone): boolean
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { isBefore } from '@/datezone';

const date1 = new Date(2024, 0, 15).getTime();
const date2 = new Date(2024, 0, 20).getTime();

console.log(isBefore(date1, date2)); // true (Jan 15 is before Jan 20)
console.log(isBefore(date2, date1)); // false

// Using date objects
const dateObj1 = { year: 2024, month: 1, day: 15 };
const dateObj2 = { year: 2024, month: 1, day: 20 };
console.log(isBefore(dateObj1, dateObj2)); // true
```

</Tab>
<Tab value="Timezone">

```typescript
import { isBefore } from '@/datezone';

const date1 = new Date(2024, 0, 15).getTime();
const dateObj2 = { year: 2024, month: 1, day: 20 };

// Mixed types require timezone
const isBeforeNY = isBefore(date1, dateObj2, 'America/New_York');
const isBeforeTokyo = isBefore(date1, dateObj2, 'Asia/Tokyo');

console.log(isBeforeNY); // true (comparing with timezone conversion)
console.log(isBeforeTokyo); // true
```

</Tab>
</Tabs>

## isSameDay
