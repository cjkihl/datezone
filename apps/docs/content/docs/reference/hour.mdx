# Hour Utilities

Hour utilities provide functions for working with hours, minutes, seconds, and milliseconds in dates and timestamps, including time arithmetic and time boundaries.

## get12Hour

Get the hour in 12-hour format (1-12).

```typescript
function get12Hour(ts: OptionsOrTimestamp, timeZone?: TimeZone): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { get12Hour } from '@/datezone';

const timestamp = new Date(2024, 0, 15, 14, 30).getTime(); // 2:30 PM
const hour12 = get12Hour(timestamp);
console.log(hour12); // 2

const morningTime = new Date(2024, 0, 15, 0, 30).getTime(); // 12:30 AM
const morningHour = get12Hour(morningTime);
console.log(morningHour); // 12
```

</Tab>
<Tab value="Timezone">

```typescript
import { get12Hour } from '@/datezone';

const timestamp = new Date(2024, 0, 15, 14, 30).getTime(); // 2:30 PM local
const hour12NY = get12Hour(timestamp, 'America/New_York');
const hour12Tokyo = get12Hour(timestamp, 'Asia/Tokyo');
console.log(hour12NY); // Hour in 12-format for NY timezone
console.log(hour12Tokyo); // Hour in 12-format for Tokyo timezone
```

</Tab>
</Tabs>

## addMilliseconds

Add milliseconds to a timestamp.

```typescript
function addMilliseconds(ts: OptionsOrTimestamp, amount: number): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { addMilliseconds } from '@/datezone';

const timestamp = new Date(2024, 0, 15, 14, 30, 15, 500).getTime(); // 2:30:15.500 PM
const futureTime = addMilliseconds(timestamp, 250);
console.log(new Date(futureTime)); // 2:30:15.750 PM same day

const pastTime = addMilliseconds(timestamp, -1000); // 1 second
console.log(new Date(pastTime)); // 2:30:14.500 PM same day
```

</Tab>
<Tab value="Timezone">

```typescript
import { addMilliseconds } from '@/datezone';

// Note: addMilliseconds doesn't take timezone parameter as it operates on raw timestamps
const timestamp = new Date(2024, 0, 15, 14, 30, 15, 500).getTime();
const futureTime = addMilliseconds(timestamp, 5000); // 5 seconds
console.log(new Date(futureTime)); // 2:30:20.500 PM

// For timezone-aware operations, combine with other functions
const futureInTZ = addSeconds(futureTime, 0, 'America/New_York');
```

</Tab>
