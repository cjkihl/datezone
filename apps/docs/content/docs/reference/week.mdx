# Week Utilities

Week utilities provide functions for working with weeks in dates and timestamps, including week calculations, ISO week handling, and week boundaries.

## getWeek

Get the ISO week number of the year (1-53).

```typescript
function getWeek(ts: OptionsOrTimestamp, timeZone?: TimeZone): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { getWeek } from '@/datezone';

const timestamp = new Date(2024, 0, 15).getTime(); // January 15, 2024
const weekNumber = getWeek(timestamp);
console.log(weekNumber); // 3

// Using date options
const weekFromOptions = getWeek({ year: 2024, month: 1, day: 1 });
console.log(weekFromOptions); // 1
```

</Tab>
<Tab value="Timezone">

```typescript
import { getWeek } from '@/datezone';

const timestamp = new Date(2024, 0, 1, 2, 0).getTime(); // January 1, 2024 2:00 AM
const weekInNY = getWeek(timestamp, 'America/New_York');
const weekInTokyo = getWeek(timestamp, 'Asia/Tokyo');
console.log(weekInNY); // 1
console.log(weekInTokyo); // 1 (same week, different timezone)
```

</Tab>
</Tabs>

## getISOWeekYear

Get the ISO week-numbering year.

```typescript
function getISOWeekYear(ts: OptionsOrTimestamp, timeZone?: TimeZone): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { getISOWeekYear } from '@/datezone';

const timestamp = new Date(2024, 0, 1).getTime(); // January 1, 2024
const isoYear = getISOWeekYear(timestamp);
console.log(isoYear); // 2024

// Edge case: early January might belong to previous ISO year
const earlyJan = new Date(2023, 0, 1).getTime(); // January 1, 2023
const earlyIsoYear = getISOWeekYear(earlyJan);
console.log(earlyIsoYear); // 2022 (belongs to last week of 2022)
```

</Tab>
<Tab value="Timezone">

```typescript
import { getISOWeekYear } from '@/datezone';

const timestamp = new Date(2023, 0, 1, 2, 0).getTime(); // January 1, 2023
const isoYearNY = getISOWeekYear(timestamp, 'America/New_York');
const isoYearTokyo = getISOWeekYear(timestamp, 'Asia/Tokyo');
console.log(isoYearNY); // 2022
console.log(isoYearTokyo); // 2022
```

</Tab>
</Tabs>

## startOfWeek

Get the timestamp for the start of the week.

```typescript
function startOfWeek(
  date: OptionsOrTimestamp, 
  timeZone?: TimeZone, 
  weekStartsOn: WeekStartsOn = WeekStartsOn.MONDAY
): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { startOfWeek, WeekStartsOn } from '@/datezone';

const timestamp = new Date(2024, 0, 15, 14, 30).getTime(); // Monday, January 15, 2024 2:30 PM
const weekStart = startOfWeek(timestamp);
console.log(new Date(weekStart)); // Monday, January 15, 2024 00:00:00

// Start week on Sunday
const weekStartSunday = startOfWeek(timestamp, undefined, WeekStartsOn.SUNDAY);
console.log(new Date(weekStartSunday)); // Sunday, January 14, 2024 00:00:00
```

</Tab>
<Tab value="Timezone">

```typescript
import { startOfWeek, WeekStartsOn } from '@/datezone';

const timestamp = new Date(2024, 0, 15).getTime();
const weekStartNY = startOfWeek(timestamp, 'America/New_York');
const weekStartTokyo = startOfWeek(timestamp, 'Asia/Tokyo', WeekStartsOn.SUNDAY);

console.log(new Date(weekStartNY)); // Start of week in NY timezone
console.log(new Date(weekStartTokyo)); // Start of week (Sunday) in Tokyo timezone
```

</Tab>
</Tabs>

## endOfWeek

Get the timestamp for the end of the week.

```typescript
function endOfWeek(
  date: OptionsOrTimestamp, 
  timeZone?: TimeZone, 
  weekStartsOn: WeekStartsOn = WeekStartsOn.MONDAY
): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { endOfWeek, WeekStartsOn } from '@/datezone';

const weekEnd = endOfWeek(timestamp);
console.log(new Date(weekEnd)); // Sunday, January 21, 2024 23:59:59.999

// End week on Saturday (when starting on Sunday)
const weekEndSaturday = endOfWeek(timestamp, undefined, WeekStartsOn.SUNDAY);
console.log(new Date(weekEndSaturday)); // Saturday, January 20, 2024 23:59:59.999
```

</Tab>

```typescript
console.log(new Date(weekEndNY)); // End of week in NY timezone
console.log(new Date(weekEndTokyo)); // End of week (Saturday) in Tokyo timezone
```

</Tab>

## addWeeks

Add weeks to a date.

```typescript
function addWeeks(date: OptionsOrTimestamp, amount: number, timeZone?: TimeZone): number
```

<Tabs items={["Local", "Timezone"]}>
<Tab value="Local">

```typescript
import { addWeeks } from '@/datezone';

const timestamp = new Date(2024, 0, 15).getTime(); // January 15, 2024
const futureDate = addWeeks(timestamp, 3);
console.log(new Date(futureDate)); // February 5, 2024

const pastDate = addWeeks(timestamp, -2);
console.log(new Date(pastDate)); // January 1, 2024
```

</Tab>
<Tab value="Timezone">

```typescript

const timestamp = new Date(2024, 0, 15).getTime();
const futureInNY = addWeeks(timestamp, 4, 'America/New_York');
const futureInTokyo = addWeeks(timestamp, 4, 'Asia/Tokyo');

console.log(new Date(futureInNY)); // February 12, 2024 in NY timezone
console.log(new Date(futureInTokyo)); // February 12, 2024 in Tokyo timezone
```

</Tab>
</Tabs>

## WeekStartsOn Enum

Enum for specifying which day the week starts on.

```typescript
enum WeekStartsOn {
  SUNDAY = 0,
  MONDAY = 1,
  SATURDAY = 6
}
```

```typescript
import { WeekStartsOn } from '@/datezone';

// Use in functions that accept weekStartsOn parameter
const weekStart = startOfWeek(timestamp, 'UTC', WeekStartsOn.SUNDAY);
const weekEnd = endOfWeek(timestamp, 'UTC', WeekStartsOn.MONDAY);
const weeksCount = getWeeksInMonth(timestamp, 'UTC', WeekStartsOn.SATURDAY);
```
