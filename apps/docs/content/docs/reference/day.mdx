### weekDayName

Returns the localized weekday name.

```ts
weekDayName(locale: string, type: "long" | "short" | "narrow", day: number): string
```

<Tabs items={["en-US", "fr-FR"]}>
  <Tab value="en-US">

```ts title="en-US"

```
  </Tab>
  <Tab value="fr-FR">

```ts title="fr-FR"

```
  </Tab>
</Tabs>
